  #!/bin/python3

  import sys

  def insertionSort1(start, arr):
      probe = arr[start]

      for ind in range(start-1, -1, -1):
          if arr[ind] > probe:
              arr[ind+1] = arr[ind]
          else:
              arr[ind+1] = probe
              break
      if arr[0] > probe:
          arr[0] = probe

  def insertionSort2(n, arr):

          insertionSort1(ind, arr)
          print(" ".join(map(str, arr)))
