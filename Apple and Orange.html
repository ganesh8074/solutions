  #!/bin/python3

  import sys

  def appleAndOrange(s, t, a, b, apple, orange):
      count_app = 0
      count_org = 0
      for el in apple:
          if s <= el+a <= t:
              count_app += 1

      for el in orange:
          if s <= el+b <= t:
              count_org += 1

      return [ count_app, count_org ]
