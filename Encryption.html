  #!/bin/python3

  import sys
  from math import sqrt
  from math import ceil

  def get_grid(number):
      root = sqrt(number)

      x = int(root//1)
      y = ceil(root)

      while x*y < number:
          if x <= y:
              x += 1
          else:
              y += 1

      return (x, y)
