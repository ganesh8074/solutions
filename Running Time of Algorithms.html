  #!/bin/python3

  import sys

  def insertionSort1(start, arr):
      probe = arr[start]
      changed = 0

      for ind in range(start-1, -1, -1):
          if arr[ind] > probe:
              changed += 1
              arr[ind+1] = arr[ind]
          else:
              arr[ind+1] = probe
              break
      if arr[0] > probe:
          arr[0] = probe

      return changed

  def insertionSort2(n, arr):
      res = 0
      for ind in range(1, len(arr)):
